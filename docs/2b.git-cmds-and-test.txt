# --- Git Commands for Version Control ---

# Initialize a new Git repository
git init

# Add remote 
git remote add origin https://github.com/cprakashb/labc-test.git

# Stage all project files for commit
git add .

# First commit with a clear message
git commit -m "Created LABC Assignment demo"

# Push to main branch
git branch -M main
git push -u origin main

# For example: Add changes after code review or bug fix
git add posts.js
git commit -m "Fix: handle missing file gracefully and improve MIME type check"
git push


# --- CI/CD Configuration (using GitHub Actions as an example) ---

# File: .github/workflows/ci.yml
name: LABC CI/CD Pipeline

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  build-test-lint:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout Code
        uses: actions/checkout@v3

      - name: Set up Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '20'

      - name: Remove node_modules and lockfile
        run: |
          rm -rf node_modules package-lock.json
          npm cache clean --force

      - name: Install Dependencies Fresh
        run: npm install

      - name: Run Linter
        run: npm run lint

      - name: Run Tests
        run: npm run test

      - name: Run Build
        run: npm run build
